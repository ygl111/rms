# C++ Parser 项目依赖清单
# 这是一个 C++ 项目，以下是需要安装的系统依赖和库

# ========================================
# 系统依赖 (Ubuntu/Debian)
# ========================================

# 基础编译工具
build-essential
cmake>=3.10
pkg-config

# Boost 库 (异步网络和线程)
libboost-system-dev>=1.71.0
libboost-thread-dev>=1.71.0
libboost-filesystem-dev>=1.71.0

# 日志库（异步日志）
libspdlog-dev

# Redis 客户端库
libhiredis-dev

# 注意: redis++ 需要手动从源码编译安装
# 请先运行: ./install_redis_plus_plus.sh
# 或者手动安装，见下方命令

# MySQL 客户端库
libmysqlclient-dev
libmysqlcppconn-dev

# RabbitMQ 客户端库 (rabbitmq-c)
librabbitmq-dev

# 线程库
libpthread-stubs0-dev

# JSON 库 (项目已包含 external/include/json.hpp)
# 无需安装系统包

# SSL 库 (OpenSSL)
libssl-dev

# ========================================
# 可选依赖
# ========================================

# 调试工具
gdb
valgrind

# 性能分析工具
perf

# 网络工具
net-tools
iproute2

# MySQL 服务器 (用于本地测试)
mysql-server

# git clone https://github.com/sewenew/redis-plus-plus.git
# cd redis-plus-plus
# mkdir build && cd build
# cmake ..
# make -j$(nproc)
# sudo make install
# sudo ldconfig
